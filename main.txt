import java.util.Scanner;

public class PrimeFactors
{
  public static void main( String args[] )
  {
    Scanner keyInput = new Scanner(System.in);
    int userNum = 0;
    String primeString = new String(" ");
    int primeCounter = 0;
    
    System.out.print("Enter a number: ");
    userNum = keyInput.nextInt();
    
    while (userNum < 0)
    {
      System.out.print("Enter a number: ");
      userNum = keyInput.nextInt();
    }
    int halfNum = userNum / 2;
        
    for (int i = 0; i < halfNum; i++)
    {
      if (i % 2 != 0 && i % 3 != 0)
      {
        for (int j = 2; j < Math.sqrt(halfNum); j++) // begin troubleshooting here
        {
          if (i % j != 0)
          {
            primeString += i;
            primeString += " ";
            primeCounter++;
          }
        }
      }
    }
    
    String [] primeArray = new String[primeCounter];
    primeArray = primeString.split(" ");
    int [] intArray = new int[primeArray.length];
    
    for (int i = 0; i < primeArray.length; i++)
    {
      intArray[i] = Integer.parseInt(primeArray[i]);
    }
    
    System.out.println("The Prime Factors are: ");
    for (int k = 0; k < intArray.length; k++)
    {
      if (userNum % intArray[k] == 0)
      {
        System.out.print(intArray[k] + " ");
      }
    } 
  } // end of main method
}
